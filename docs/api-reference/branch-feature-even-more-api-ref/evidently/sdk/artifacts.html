class ArtifactContentType(builtins.str, enum.Enum):

Type of content stored in an artifact.

Determines how the artifact content is interpreted and used.

Prompt = <ArtifactContentType.Prompt: 'prompt'>

Prompt template content.

Config = <ArtifactContentType.Config: 'config'>

Configuration content.

Descriptor = <ArtifactContentType.Descriptor: 'descriptor'>

Descriptor definition content.

RunDescriptorsConfig = <ArtifactContentType.RunDescriptorsConfig: 'run-descriptors-config'>

Run descriptors configuration content.

class ArtifactContent(evidently.pydantic_utils.AutoAliasMixin, evidently.pydantic_utils.EvidentlyBaseModel, typing.Generic[~TArtifactValue], abc.ABC):

Base class for artifact content wrappers.

Artifact content wraps typed values (e.g., prompts, configs) for storage and versioning in the artifact system.

class ArtifactMetadata(pydantic.v1.main.BaseModel):

Metadata for an artifact.

Stores creation/update timestamps, author information, and optional description.

created_at: datetime.datetime

Creation timestamp.

updated_at: datetime.datetime

Last update timestamp.

author: Optional[uuid.UUID]

Optional user ID of the creator.

description: Optional[str]

Optional description of the artifact.

class Artifact(pydantic.v1.main.BaseModel):

An artifact in the Evidently system.

Artifacts are versioned objects that can store prompts, configs, descriptors, or other structured data. Each artifact has a name, metadata, and multiple versions.

id: uuid.UUID

Unique artifact identifier.

project_id: uuid.UUID

Project this artifact belongs to.

name: str

Name of the artifact.

metadata: ArtifactMetadata

Metadata about the artifact.

class ArtifactVersionMetadata(pydantic.v1.main.BaseModel):

Metadata for an artifact version.

Stores creation/update timestamps, author information, and optional comment describing the changes in this version.

created_at: datetime.datetime

Creation timestamp.

updated_at: datetime.datetime

Last update timestamp.

author: Optional[uuid.UUID]

Optional user ID of the creator.

comment: Optional[str]

Optional comment describing this version.

class ArtifactVersion(pydantic.v1.main.BaseModel):

A version of an artifact.

Each artifact can have multiple versions, allowing you to track changes over time and roll back if needed. Versions are numbered sequentially starting from 1.

ArtifactVersion( content: Union[Any, ArtifactContent], version: int, id: uuid.UUID = UUID('00000000-0000-7000-8000-000000000000'), artifact_id: uuid.UUID = UUID('00000000-0000-7000-8000-000000000000'), metadata: Optional[ArtifactVersionMetadata] = None, content_type: Optional[ArtifactContentType] = None, **data: Any)

Create a new model by parsing and validating input data from keyword arguments.

class RemoteArtifact(Artifact):

Remote artifact with API access.

Provides methods to interact with a remote artifact through the API, including version management and updates. Use ArtifactAPI to create and manage remote artifacts.

id: uuid.UUID

Unique artifact identifier.

project_id: uuid.UUID

Project this artifact belongs to.

name: str

Name of the artifact.

metadata: ArtifactMetadata

Metadata about the artifact.

def delete(self):

Delete this artifact and all its versions.

This operation is irreversible and will permanently remove the artifact and all associated versions from the workspace.

def delete_version(self, version_id: uuid.UUID):

Delete a specific version.

This operation is irreversible and will permanently remove the version.

Args:

  • version_id: ID of the version to delete.
def save(self):

Save changes to this artifact's metadata to the remote workspace.

Updates the artifact's name and metadata fields. Does not affect versions.
